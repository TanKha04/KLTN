\section{CHI TIẾT CHỨC NĂNG CÁC FILE QUAN TRỌNG}
% ============================================================

\subsection{config.php - File cấu hình trung tâm}

\textbf{Chức năng Backend:}
\begin{itemize}
    \item Kết nối database MySQL qua PDO
    \item Quản lý session (session\_start)
    \item Cấu hình SMTP để gửi email
    \item Cấu hình Facebook OAuth
    \item Các hàm tiện ích:
    \begin{itemize}
        \item \texttt{is\_logged\_in()}: Kiểm tra đăng nhập
        \item \texttt{require\_login()}: Bắt buộc đăng nhập
        \item \texttt{is\_admin\_user()}: Kiểm tra quyền admin
        \item \texttt{require\_admin()}: Bắt buộc quyền admin
        \item \texttt{is\_student\_verified()}: Kiểm tra sinh viên đã xác thực
        \item \texttt{is\_user\_online()}: Kiểm tra trạng thái online
        \item \texttt{ensure\_schema\_updates()}: Tự động cập nhật cấu trúc DB
    \end{itemize}
\end{itemize}

\subsection{header.php - Thanh điều hướng}

\textbf{Chức năng Frontend:}
\begin{itemize}
    \item Navbar responsive với Bootstrap 5
    \item Logo và tên ứng dụng
    \item Menu điều hướng động theo vai trò (patient/student/admin)
    \item Dropdown menu người dùng với avatar
    \item Hiển thị trạng thái online
    \item Link đến các chức năng: Dashboard, Chat, Hồ sơ, Đăng xuất
\end{itemize}

\subsection{index.php - Trang chủ}

\textbf{Backend:}
\begin{itemize}
    \item Truy vấn danh sách bài đăng từ database
    \item Xử lý tìm kiếm theo: từ khóa (q), loại tin (type), danh mục (category), khu vực (area)
    \item Join bảng users để lấy thông tin tác giả
\end{itemize}

\textbf{Frontend:}
\begin{itemize}
    \item Hero section với hình ảnh và thống kê
    \item Form tìm kiếm/lọc tin
    \item Grid hiển thị danh sách bài đăng (card design)
    \item Hiển thị trạng thái online của tác giả
    \item Responsive design cho mobile
\end{itemize}

\subsection{login.php - Đăng nhập}

\textbf{Backend:}
\begin{itemize}
    \item Xác thực email/password với password\_verify()
    \item Kiểm tra email đã xác thực chưa
    \item Tạo session với thông tin user
    \item Cập nhật last\_login
    \item Redirect theo vai trò
\end{itemize}

\textbf{Frontend:}
\begin{itemize}
    \item Form đăng nhập với validation
    \item Link đăng nhập Facebook
    \item Link quên mật khẩu
    \item Link đăng ký
\end{itemize}

\subsection{register.php - Đăng ký}

\textbf{Backend:}
\begin{itemize}
    \item Validate dữ liệu đầu vào
    \item Kiểm tra email trùng lặp
    \item Hash password với password\_hash()
    \item Tự động phân quyền: email .edu.vn = student, còn lại = patient
    \item Gửi email xác thực
    \item Tạo bản ghi trong bảng users
\end{itemize}

\subsection{chat.php - Hệ thống chat}

\textbf{Backend:}
\begin{itemize}
    \item Tạo/lấy conversation giữa 2 user
    \item Lưu tin nhắn vào bảng direct\_messages
    \item Truy vấn lịch sử tin nhắn
\end{itemize}

\textbf{Frontend:}
\begin{itemize}
    \item Giao diện chat 2 cột (danh sách + nội dung)
    \item Hiển thị avatar và trạng thái online
    \item Form gửi tin nhắn
    \item Auto-scroll đến tin mới nhất

\subsection{admin\_dashboard.php - Bảng điều khiển Admin}

\begin{itemize}
    \item Thống kê: tổng users, posts, messages, reports

\section{CÁC TÍNH NĂNG CHÍNH CỦA HỆ THỐNG}

\subsection{Tính năng xác thực và phân quyền}

\begin{enumerate}
    \item \textbf{Đăng ký/Đăng nhập:}
    \begin{itemize}
        \item Đăng ký bằng email thường hoặc email sinh viên
        \item Đăng nhập bằng Facebook OAuth
        \item Xác thực email qua token
        \item Quên mật khẩu/Reset password
    \end{itemize}
    
    \item \textbf{Phân quyền 3 cấp:}
    \begin{itemize}
        \item \textbf{Patient (Bệnh nhân):} Đăng tin tuyển dụng, tìm sinh viên
        \item \textbf{Student (Sinh viên):} Đăng tin ứng tuyển (sau khi xác thực)
        \item \textbf{Admin:} Quản lý toàn bộ hệ thống
    \end{itemize}
    
    \item \textbf{Xác thực sinh viên:}
    \begin{itemize}
        \item Upload ảnh thẻ sinh viên
        \item Admin duyệt/từ chối
        \item Cấp quyền đăng tin sau khi xác thực
    \end{itemize}
\end{enumerate}

\subsection{Tính năng quản lý bài đăng}

\begin{enumerate}
    \item \textbf{Tạo bài đăng:}
    \begin{itemize}
        \item Tin tuyển dụng (recruitment) - cho bệnh nhân
        \item Tin ứng tuyển (application) - cho sinh viên
        \item Nhập: tiêu đề, nội dung, danh mục, khu vực, giá đề xuất, thông tin liên hệ
    \end{itemize}
    
    \item \textbf{Quản lý trạng thái:}
    \begin{itemize}
        \item Open: Đang mở
        \item Taken: Đã có người nhận
        \item Closed: Đã đóng
    \end{itemize}
    
    \item \textbf{Tìm kiếm và lọc:}
    \begin{itemize}
        \item Tìm theo từ khóa
        \item Lọc theo loại tin
        \item Lọc theo khu vực
        \item Lọc theo danh mục/chuyên khoa
    \end{itemize}
\end{enumerate}

\subsection{Tính năng giao tiếp}

\begin{enumerate}
    \item \textbf{Hệ thống tin nhắn:}
    \begin{itemize}
        \item Chat trực tiếp giữa 2 người dùng
        \item Tin nhắn liên quan đến bài đăng
        \item Lưu trữ lịch sử hội thoại

    \item \textbf{Hệ thống bạn bè:}
    \begin{itemize}

        \item Chấp nhận/Từ chối lời mời

    \end{itemize}
    
    \item \textbf{Bình luận:}
    \begin{itemize}

        \item Trả lời bình luận (nested comments)
    \end{itemize}
\end{enumerate}

\subsection{Tính năng đánh giá và báo cáo}

\begin{enumerate}
    \item \textbf{Đánh giá người dùng:}
    \begin{itemize}
        \item Đánh giá 1-5 sao
        \item Viết nhận xét
        \item Hiển thị điểm trung bình
    \end{itemize}
    
    \item \textbf{Báo cáo vi phạm:}
    \begin{itemize}
        \item Báo cáo bài đăng
        \item Báo cáo bình luận
        \item Chọn lý do vi phạm
        \item Admin xử lý báo cáo
    \end{itemize}
\end{enumerate}

\subsection{Tính năng yêu thích}

\begin{itemize}
    \item Lưu bài đăng vào danh sách yêu thích
    \item Xem danh sách bài đăng đã lưu
    \item Xóa khỏi danh sách yêu thích

\subsection{Tính năng quản trị (Admin)}

\begin{enumerate}
    \item \textbf{Dashboard:} Thống kê tổng quan hệ thống
    \item \textbf{Quản lý người dùng:} Xem, khóa, xóa, cấp quyền

\subsection{Tính năng UX/UI}

\begin{enumerate}

    \item \textbf{Avatar:} Upload và hiển thị ảnh đại diện

\section{SƠ ĐỒ LUỒNG DỮ LIỆU}
% ============================================================

\subsection{Luồng đăng ký tài khoản}

\begin{verbatim}
User -> register.php -> Validate input
                     -> Check email exists
                     -> Hash password
                     -> Determine role (patient/student)
                     -> Insert to users table
                     -> Send verification email
                     -> Redirect to login
\end{verbatim}

\subsection{Luồng đăng tin}

\begin{verbatim}
User -> create_recruitment.php / create_application.php
     -> Check login & permission
     -> Validate input
     -> Insert to posts table
     -> Redirect to view_post.php
\end{verbatim}

\subsection{Luồng xác thực sinh viên}

\begin{verbatim}
Student -> request_verification.php
        -> Upload student card image
        -> Insert to verifications table (status=pending)

Admin -> admin_verifications.php
      -> Review request
      -> Approve/Reject
      -> Update users.verified = 1 (if approved)
      -> Student can now post
\end{verbatim}
